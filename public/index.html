API — One line change

# Replace OpenAI's base URL with Cocapn
# Everything else stays the same

# Python
openai.base_url = "https://cocapn.ai/v1"
openai.api_key = "cocapn_your_key_here"
response = openai.chat.completions.create(
  model="deepseek-chat",
  messages=[{"role":"user","content":"Hello!"}]
)

# cURL
curl https://cocapn.ai/v1/chat/completions \\
  -H "Authorization: Bearer cocapn_your_key" \\
  -H "Content-Type: application/json" \\
  -d '{"model":"deepseek-chat","messages":[...]}'

# Response includes cost headers:
# X-Cocapn-Cost: 0.000042
# X-Cocapn-Model: deepseek-chat
# X-Cocapn-Tokens: 15,47
